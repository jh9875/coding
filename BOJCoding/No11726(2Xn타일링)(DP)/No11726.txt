import java.util.Scanner;

public class Main
{
	public static void main(String[] args)
	{
		Scanner input =new Scanner(System.in);
		int[] arr =new int[1001];
		arr[1] =1;
		arr[2] =2;
		for(int i =3; i <=1000; i++)
			arr[i] =(arr[i -1] +arr[i -2]) %10007;
		int n =input.nextInt();
		System.out.println(arr[n]);
	}
	
}
