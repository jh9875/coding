import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.Arrays;
import java.util.PriorityQueue;
import java.util.StringTokenizer;

public class Main
{
	static class Pos implements Comparable<Pos>
	{
		int y;
		int x;
		int cost;
		int wall;
		
		Pos(int y, int x, int cost, int wall)
		{
			this.y =y;
			this.x =x;
			this.cost =cost;
			this.wall =wall;
		}

		@Override
		public int compareTo(Pos p)
		{
			return this.cost -p.cost;
		}
	}
	static int N, M;
	static int[][] map;
	static int[] dirY = {1, 0, -1, 0};
	static int[] dirX = {0, 1, 0, -1};
	
	public static void main(String[] args) throws Exception
	{
		BufferedReader br =new BufferedReader(new InputStreamReader(System.in));
		BufferedWriter bw =new BufferedWriter(new OutputStreamWriter(System.out));
		StringTokenizer st;
		String input;
		int ans =0;
		
		st =new StringTokenizer(br.readLine());
		N =Integer.parseInt(st.nextToken());
		M =Integer.parseInt(st.nextToken());
		map =new int[N+1][M+1];
		
		for(int i =0; i <N; i++)
		{
			input =br.readLine();
			for(int j =0; j <M; j++)
				map[i+1][j+1] =input.charAt(j) -'0';
		}
		
		ans =bfsSearch();
		bw.write(ans +"\n");
		bw.flush();
	}
	
	static int bfsSearch()
	{
		int ret =-1;
		int[][] dist =new int[N+1][M+1];
		for(int i =1; i <=N; i++)
			Arrays.fill(dist[i], Integer.MAX_VALUE);
		PriorityQueue<Pos> pq =new PriorityQueue<>();
		pq.add(new Pos(1, 1, 1, 0));
		dist[1][1] =1;
		
		while(!pq.isEmpty())
		{
			int y =pq.peek().y;
			int x =pq.peek().x;
			int cost =pq.peek().cost;
			int wall =pq.peek().wall;
			pq.poll();
			
			if(y ==N && x ==M)
			{
				ret =cost;
				break;
			}
			
			if(dist[y][x] <cost)
				continue;
			
			for(int i =0; i <4; i++)
			{
				int nextY =y +dirY[i];
				int nextX =x +dirX[i];
				int nextCost =cost+1;
				int nextWall =wall;
				
				if(inRange(nextY, nextX))
				{
					if(map[nextY][nextX] ==1)
						nextWall =wall+1;
					if(nextWall >1)
						continue;
					if(dist[nextY][nextX] >nextCost)
					{
						dist[nextY][nextX] =nextCost;
						pq.add(new Pos(nextY, nextX, nextCost, nextWall));
					}
					
				}
			}
		}
		return ret;
	}
	
	static boolean inRange(int y, int x)
	{
		if(y <1 || N <y || x <1 || M <x)
			return false;
		return true;
	}
}
